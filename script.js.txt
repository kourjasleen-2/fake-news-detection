function checkNews() {
    let input = document.getElementById("newsInput").value.toLowerCase();
    let result = document.getElementById("result");

    if (input.trim() === "") {
        result.innerText = "Please enter some news text.";
        result.style.color = "orange";
        return;
    }

    let fakeWords = ["shocking", "click here", "100% cure", "secret", "guaranteed"];
    let isFake = fakeWords.some(word => input.includes(word));

    if (isFake) {
        result.innerText = "This news is likely FAKE!";
        result.style.color = "red";
    } else {
        result.innerText = "This news seems REAL.";
        result.style.color = "green";
    }
}
